323  if(debug_messages) {
324  std::cerr << "feedback_fn=" << pty_fn << std::endl;
325  }
326 
327  // disable echo
328  struct termios tios;
329  if(tcgetattr(slave_fd, &tios) < 0) {
330  perror("tcgetattr");
331  throw std::runtime_error("tcgetattr failed");
332  }
333  tios.c_lflag &= ~(ECHO | ECHONL);
334  if(tcsetattr(slave_fd, TCSAFLUSH, &tios) < 0) {
335  perror("tcsetattr");
336  throw std::runtime_error("tcsetattr failed");
337  }
338

341  throw std::runtime_error("fdopen failed");
342  }
343  }

412 // {{{1 Traits and printers for entry datatypes
413 //
414 // This section contains the mechanisms for sending scalar and tuple data to gnuplot. Pairs
415 // and tuples are sent by appealing to the senders defined for their component scalar types.
416 // Senders for arrays are defined in a later section.
417 //
418 // There are three classes which need to be specialized for each supported datatype:
419 // 1. TextSender to send data as text. The default is to just send using the ostream's `<<`
420 // operator.
421 // 2. BinarySender to send data as binary, in a format which gnuplot can understand. There is
422 // no default implementation (unimplemented types raise a compile time error), however
423 // inheriting from FlatBinarySender will send the data literally as it is stored in memory.
424 // This suffices for most of the standard built-in types (e.g. uint32_t or double).
425 // 3. BinfmtSender sends a description of the data format to gnuplot (e.g. `%uint32`). Type
426 // `show datafile binary datasizes` in gnuplot to see a list of supported formats.
427 
428 // {{{2 Basic entry datatypes

791 // {{{1 ArrayTraits and Range classes
792 //
793 // This section handles sending of array data to gnuplot. It is rather complicated because of
794 // the diversity of storage schemes supported. For example, it treats a
795 // `std::pair<std::vector<T>, std::vector<U>>` in the same way as a
796 // `std::vector<std::pair<T, U>>`, iterating through the two arrays in lockstep, and sending
797 // pairs <T,U> to gnuplot as columns. In fact, any nested combination of pairs, tuples, STL
798 // containers, Blitz arrays, and Armadillo arrays is supported (with the caveat that, for
799 // instance, Blitz arrays should never be put into an STL container or you will suffer
800 // unpredictable results due the way Blitz handles assignment). Nested containers are
801 // considered to be multidimensional arrays. Although gnuplot only supports 1D and 2D arrays,
802 // our module is in principle not limited.
803 //
804 // The ArrayTraits class is specialized for every supported array or container type (the
805 // default, unspecialized, version of ArrayTraits exists only to tell you that something is
806 // *not* a container, via the is_container flag). ArrayTraits tells you the depth of a nested
807 // (or multidimensional) container, as well as the value type, and provides a specialized
808 // sort of iterator (a.k.a. "range"). Ranges are sort of like STL iterators, except that they
809 // have built-in knowledge of the end condition so you don't have to carry around both a
810 // begin() and an end() iterator like in STL.
811 //
812 // As an example of how this works, consider a std::pair of std::vectors. Ultimately this gets
813 // sent to gnuplot as two columns, so the two vectors need to be iterated in lockstep.
814 // The `value_type` of `std::pair<std::vector<T>, std::vector<U>>` is then `std::pair<T, U>`
815 // and this is what deferencing the range (iterator) gives. Internally, this range is built
816 // out of a pair of ranges (PairOfRange class), the `inc()` (advance to next element)
817 // method calls `inc()` on each of the children, and `deref()` calls `deref()` on each child
818 // and combines the results to return a `std::pair`. Tuples are handled as nested pairs.
819 //
820 // In addition to PairOfRange, there is also a VecOfRange class that can be used to treat the
821 // outermost part of a nested container as if it were a tuple. Since tuples are printed as
822 // columns, this is like treating a multidimensional array as if it were column-major. A
823 // VecOfRange is obtained by calling `get_columns_range`. This is used by, for instance,
824 // `send1d_colmajor`. The implementation is similar to that of PairOfRange.
825 //
826 // The range, accessed via `ArrayTraits<T>::get_range`, will be of a different class depending
827 // on T, and this is defined by the ArrayTraits specialization for T. It will always have
828 // methods `inc()` to advance to the next element and `is_end()` for checking whether one has
829 // advanced past the final element. For nested containers, `deref_subiter()` returns a range
830 // iterator for the next nesting level. When at the innermost level of nesting, `deref()`
831 // returns the value of the entry the iterator points to (a scalar, pair, or tuple).
832 // Only one of `deref()` or `deref_subiter()` will be available, depending on whether there are
833 // deeper levels of nesting. The typedefs `value_type` and `subiter_type` tell the return
834 // types of these two methods.
835 //
836 // Support for standard C++ and boost containers and tuples of containers is provided in this
837 // section. Support for third party packages like Blitz and Armadillo is in a later section.
838

1165  return true;

1451 // {{{2 print_block()
1452 //
1453 // Here we format nested containers into blocks (for 2D data) and lines. Actually, block and
1454 // line formatting is only truely needed for text mode output, but for uniformity this function
1455 // is also invoked in binary mode (the PrintMode tag determines the output mode). If the goal
1456 // is to just print the array size or the binary format string, then the loops exit after the
1457 // first iteration.
1458 //
1459 // The Depth argument tells how deep to recurse. It will be either `2` for 2D data, formatted
1460 // into blocks and lines, with empty lines between blocks, or `1` for 1D data formatted into
1461 // lines but not blocks. Gnuplot only supports 1D and 2D data, but if it were to support 3D in
1462 // the future (e.g. volume rendering), all that would be needed would be some trivial changes
1463 // in this section. After Depth number of nested containers have been recursed into, control
1464 // is passed to deref_and_print(), which treats any further nested containers as columns.
1465 
1466 // Depth==1 and we are not asked to print the size of the array. Send each element of the
1467 // range to deref_and_print() for further processing into columns.
1468 template<size_t Depth, typename T, typename PrintMode>
1469 typename boost::enable_if_c<(Depth == 1) && !PrintMode::is_size>::type print_block(
1470  std::ostream &stream, T &arg, PrintMode) {
1471  if (PrintMode::is_binfmt && arg.is_end())
1472  throw plotting_empty_container();
1473  for (; !arg.is_end(); arg.inc()) {
1474  //print_entry(arg.deref());
1475  deref_and_print(stream, arg, PrintMode());
1476  // If asked to print the binary format string, only the first element needs to be
1477  // looked at.
1478  if (PrintMode::is_binfmt)
1479  break;
1480  if (PrintMode::is_text)
1481  stream << std::endl;
1482  }
1483 }

1485 // Depth>1 and we are not asked to print the size of the array. Loop over the range and
1486 // recurse into print_block() with Depth -> Depth-1.
1487 template<size_t Depth, typename T, typename PrintMode>
1488 typename boost::enable_if_c<(Depth > 1) && !PrintMode::is_size>::type print_block(
1489  std::ostream &stream, T &arg, PrintMode) {
1490  if (PrintMode::is_binfmt && arg.is_end())
1491  throw plotting_empty_container();

1493  for (; !arg.is_end(); arg.inc()) {
1494  if (first) {
1495  first = false;
1496  } else {
1497  if (PrintMode::is_text)
1498  stream << std::endl;
1499  }

1501  stream << "<block>" << std::endl;
1502  if (arg.is_end())
1503  throw plotting_empty_container();
1504  typename T::subiter_type sub = arg.deref_subiter();
1505  print_block<Depth - 1>(stream, sub, PrintMode());
1506  // If asked to print the binary format string, only the first element needs to be
1507  // looked at.
1508  if (PrintMode::is_binfmt)
1509  break;
1510  }
1511 }
